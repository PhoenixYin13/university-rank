let eloData = {};
let totalVotes = 0;

function initElo() {
  const saved = localStorage.getItem('uniwar_elo_v2');
  const savedVotes = localStorage.getItem('uniwar_votes_v2');
  if (saved) {
    eloData = JSON.parse(saved);
    totalVotes = parseInt(savedVotes || '0');
    // Make sure any new universities added get initialized
    universities.forEach(u => {
      if (!eloData[u.id]) {
        eloData[u.id] = { elo: 1500, wins: 0, losses: 0 };
      }
    });
  } else {
    universities.forEach(u => {
      // Top QS schools start slightly higher to reflect prestige
      const bonus = Math.max(0, Math.round((30 - Math.min(u.qs, 30)) * 1.5));
      eloData[u.id] = { elo: 1500 + bonus, wins: 0, losses: 0 };
    });
  }
  document.getElementById('totalVotes').textContent = totalVotes.toLocaleString();
}

function saveElo() {
  localStorage.setItem('uniwar_elo_v2', JSON.stringify(eloData));
  localStorage.setItem('uniwar_votes_v2', totalVotes.toString());
}

function calcEloChange(winnerElo, loserElo) {
  const K = 32;
  const expected = 1 / (1 + Math.pow(10, (loserElo - winnerElo) / 400));
  return Math.round(K * (1 - expected));
}

// ============================================================
// BATTLE
// ============================================================
let currentA = null, currentB = null;
let isAnimating = false;

function getRandomPair() {
  let a, b;
  do {
    a = universities[Math.floor(Math.random() * universities.length)];
    b = universities[Math.floor(Math.random() * universities.length)];
  } while (a.id === b.id);
  return [a, b];
}

function renderBattle(uniA, uniB) {
  currentA = uniA;
  currentB = uniB;

  ['A','B'].forEach(side => {
    const u = side === 'A' ? uniA : uniB;
    document.getElementById(`name${side}`).textContent = u.name;
    document.getElementById(`flag${side}`).textContent = u.flag;
    document.getElementById(`country${side}`).textContent = u.country;
    document.getElementById(`rank${side}`).textContent = `QS RANK #${u.qs}`;
    document.getElementById(`elo${side}`).textContent = Math.round(eloData[u.id].elo);
  });
}

function vote(winner) {
  if (isAnimating) return;
  isAnimating = true;

  const winUni = winner === 'A' ? currentA : currentB;
  const loseUni = winner === 'A' ? currentB : currentA;

  const change = calcEloChange(eloData[winUni.id].elo, eloData[loseUni.id].elo);
  eloData[winUni.id].elo += change;
  eloData[winUni.id].wins++;
  eloData[loseUni.id].elo -= change;
  eloData[loseUni.id].losses++;
  totalVotes++;

  document.getElementById('totalVotes').textContent = totalVotes.toLocaleString();
  saveElo();

  // Flash result
  document.getElementById('resultName').textContent = winUni.name;
  document.getElementById('resultElo').innerHTML =
    `${winUni.flag} <span class="up">+${change} ELO</span> &nbsp;·&nbsp; ${loseUni.flag} <span class="down">-${change} ELO</span>`;

  const flash = document.getElementById('resultFlash');
  flash.classList.add('show');

  setTimeout(() => {
    flash.classList.remove('show');
    const [a, b] = getRandomPair();
    renderBattle(a, b);
    isAnimating = false;
  }, 800);
}

function nextBattle() {
  const [a, b] = getRandomPair();
  renderBattle(a, b);
}

}

function renderRankings() {
  const sorted = [...universities].sort((a, b) => eloData[b.id].elo - eloData[a.id].elo);
  const filtered = currentFilter
    ? sorted.filter(u => u.name.toLowerCase().includes(currentFilter) || u.country.toLowerCase().includes(currentFilter) || u.short.toLowerCase().includes(currentFilter))
    : sorted;

  const tbody = document.getElementById('rankingsBody');
  tbody.innerHTML = '';

  filtered.forEach((u, i) => {
    const d = eloData[u.id];
    const globalRank = sorted.indexOf(u) + 1;
    const total = d.wins + d.losses;
    const winrate = total > 0 ? Math.round((d.wins / total) * 100) : 50;

    tbody.innerHTML += `
      <tr>
        <td class="rank-num ${globalRank <= 3 ? 'top3' : ''}">${globalRank}</td>
        <td>
          <div class="uni-info">
            <span class="uni-flag-small">${u.flag}</span>
            <div>
              <div class="uni-name-cell">${u.name}</div>
              <div class="uni-country-cell">${u.country}</div>
            </div>
          </div>
        </td>
        <td style="text-align:right;font-size:12px;color:var(--muted)">${u.qs}</td>
        <td class="elo-cell">${Math.round(d.elo)}</td>
        <td class="wins-cell">${d.wins}W/${d.losses}L</td>
        <td>
          <div class="winrate-bar">
            <span style="font-size:11px;color:var(--muted)">${winrate}%</span>
            <div class="bar-bg"><div class="bar-fill" style="width:${winrate}%"></div></div>
          </div>
        </td>
      </tr>
    `;
  });
}

// ============================================================
// TABS
// ============================================================
function showTab(tab) {
  document.querySelectorAll('nav button').forEach((b, i) => {
    b.classList.toggle('active', (i === 0 && tab === 'battle') || (i === 1 && tab === 'rankings'));
  });
  document.getElementById('tab-battle').classList.toggle('active', tab === 'battle');
  document.getElementById('tab-rankings').classList.toggle('active', tab === 'rankings');
  if (tab === 'rankings') renderRankings();
}
